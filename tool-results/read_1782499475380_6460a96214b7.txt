  1901→  1901→    @staticmethod
  1902→  1902→    def calculate_vat_breakdown(selling_price_ttc):
  1903→  1903→        """
  1904→  1904→        RRA requires VAT-inclusive pricing.
  1905→  1905→        Given TTC price, extract HT and VAT.
  1906→  1906→        """
  1907→  1907→        ttc = Decimal(str(selling_price_ttc))
  1908→  1908→        ht = round(ttc * Decimal("100") / Decimal("118"), 2)
  1909→  1909→        vat = round(ttc - ht, 2)
  1910→  1910→        return {"ttc": ttc, "ht": ht, "vat": vat}
  1911→  1911→
  1912→  1912→    @classmethod
  1913→  1913→    def generate_mrc(cls, receipt_no, date_str):
  1914→  1914→        """
  1915→  1915→        MRC = Machine Receipt Code
  1916→  1916→        Deterministic hash for RRA verification
  1917→  1917→        """
  1918→  1918→        import hashlib
  1919→  1919→        raw = f"{cls.TIN}{cls.SDC_ID}{receipt_no}{date_str}"
  1920→  1920→        hash_obj = hashlib.sha256(raw.encode())
  1921→  1921→        mrc = hash_obj.hexdigest()[:16].upper()
  1922→  1922→        formatted = '-'.join([mrc[i:i+4] for i in range(0,16,4)])
  1923→  1923→        return formatted

  1925→  1925→    @classmethod
  1926→  1926→    def create_receipt(cls, order):
  1927→  1927→        """
  1928→  1928→        Creates EBM receipt with full VAT breakdown.
  1929→  1929→        Each line item shows HT + VAT + TTC separately.
  1930→  1930→        """
  1931→  1931→        from .models import EBMReceipt
  1932→  1932→        from orders.models import Order
  1933→  1933→
  1934→  1934→        receipt_no = f"RCP{order.created_at.strftime('%Y%m%d')}" \
